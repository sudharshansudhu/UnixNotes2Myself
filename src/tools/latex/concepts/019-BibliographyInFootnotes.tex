% Description: Footnotes in Latex

% Note
% 1. The abilities of BibTeX are limited to basic styles as depicted in the previous example.
% 2. Sometimes it is necessary to cite all literature in footnotes.
% 3. Use BibLaTeX instead of BibTex for footnotes.
% 4. Include the biblatex package and use the \autocite and \printbibliography command.
% 5. It is crucial to move the \bibliography{019-BibliographyInFootnotes} statement to the preamble of the document.

\documentclass{article}

% For BibLaTeX we have to choose the citation style on package.
\usepackage[backend=bibtex,style=verbose-trad2]{biblatex}
\bibliography{019-BibliographyInFootnotes}

\begin{document}

% Add page number in the brackets after \autocite
Some text needs citations \autocite[100-101]{BOOK_REF:1} embeddeed in text.

Some more text needs citation \autocite[1]{BOOK_REF:2} embeddeed in text.

Some text needs citations \autocite[1]{BOOK_REF:3} embeddeed in text.

Some text needs citations \autocite[1]{ARTICLE_REF:1} embeddeed in text.

Some text needs citations \autocite[1]{ARTICLE_REF:2} embeddeed in text.

Some text needs citations \autocite[1]{WEBSITE_REF:1} embeddeed in text.

Some text needs citations \autocite[1]{WEBSITE_REF:2} embeddeed in text.

\newpage

% This will print the complete bibliography at the end of the book but with a pdtlatex style.
\printbibliography

\end{document}
