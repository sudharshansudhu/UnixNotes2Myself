% Description: Sectioning Elements in Latex

% Note
% 1. There can be zero or more section elements.
% 2. There can be zero or more subsection elements nested inside a section element.
% 3. There can be zero or more subsubsection elements nested inside a subsection element.
% 4. There can be zero or more paragraph elements nested inside a section, subsection or subsection elements.
% 5. There can be zero or more subparagraph elements nested inside a paragraph element.
% 6. The sections, subsections and subsubsections will have successive numbers and appear in the table of contents.
% 7. Paragraphs are not numbered and thus don't appear in the table of contents.

\documentclass{article}

\begin{document}

\section{Section}

Hello World!

\subsection{Subsection}

Structuring a document is easy!

\subsubsection{Subsubsection}

More text.

\paragraph{Paragraph}

Some more text.

\subparagraph{Subparagraph}

Even more text.

\section{Another section}

\end{document}
