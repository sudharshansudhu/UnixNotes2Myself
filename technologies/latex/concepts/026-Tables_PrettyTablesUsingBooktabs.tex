% Description: Pretty Tables Using Booktabs

% Note
% 1. The default hlines used by the table environment are not very pretty. Use the booktabs package, which provides much
%    prettier horizontal separators.

\documentclass{article}

\usepackage{siunitx}
\usepackage{booktabs}                                                   % For prettier tables

\sisetup{
  round-mode          = places,     % Rounds numbers
  round-precision     = 2,          % to 2 places
}

\begin{document}

\section{Tables in \LaTeX}

\subsection{Prettier tables using booktabs}

\begin{table}[h!]
    \begin{center}
        \caption{Table using booktabs.}
        \label{tab:table1}
        \begin{tabular}{l|S|r}
            \toprule                                                    % Top Rule
            \textbf{Value 1} & \textbf{Value 2} & \textbf{Value 3}\\
            $\alpha$ & $\beta$ & $\gamma$ \\
            \midrule                                                    % Mid Line
            1 & 1110.1 & a\\
            2 & 10.1 & b\\
            3 & 23.113231 & c\\
            \bottomrule                                                 % Bottom Rule
        \end{tabular}
    \end{center}
\end{table}

\end{document}
