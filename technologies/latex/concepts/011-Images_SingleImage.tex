% Description: Images in LaTeX

% Note
% 1.  Use the graphicx package and figure environment to embed pictures.
% 2.  Pictures will be numbered automatically and tagged with successive numbers using the figure environment.
% 3.  Change the width of an image by using \includegraphics[width=\linewidth]{}.
% 5.  The \linewidth will scale (up or down) the image to fit the width of the document.
% 6.  Use \caption to show a text below the image.
% 7.  Refer to pictures in a document by setting a \label and using the \ref tag.
% 8.  Set the position of an image by adding a float option such as [h!].
% 9.  Multiple figures can be shown next to each other using the subcaption package and the subfigure environment.
% 10. For large documents, you probably want to store image files in a different folder, say we created a folder images
%     and reference using  images/file.jpg.

\documentclass{article}

\usepackage{graphicx}

\begin{document}

\begin{figure}
  \includegraphics[width=\linewidth]{011-LatexLogo.png}
  \caption{The LaTeX Logo.}
  \label{fig:logo1}
\end{figure}

Figure \ref{fig:logo1} shows a logo.

\end{document}
